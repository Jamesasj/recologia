        omega ~ dunif(0, 1)

            p[i] ~ dunif(0, 1)
        }

            z[i] ~ dbern(omega)

                yaug[i,j] ~ dbern(p.eff[i,j])
                p.eff[i,j] <- z[i] * p[j]
                } #j
            } #i
        N <- sum(z[])
    }
